UPDATE "CustomerLineItems" 
SET "CustomerLineItems"."Rate" = '{{this.params.currentRow.Rate}}', "CustomerLineItems"."Amount" = {{this.params.currentRow.Rate}} * "CustomerLineItems"."Qty"
FROM "CustomerLineItems"
INNER JOIN "Invoice"
ON "CustomerLineItems"."InvoiceID" = "Invoice"."InvoiceID"
WHERE "CustomerLineItems"."CustomerPriceID" = '{{this.params.currentRow.custprice_id}}' AND  to_date(cast("Invoice"."InvoiceDate" as TEXT), 'YYYY-MM-DD') >= '{{moment().format('YYYY-MM-DD')}}';
