SELECT Sum(c1."Qty")
FROM "public"."Invoice" i 
	INNER JOIN "public"."CustomerLineItems" c1 ON ( c1."InvoiceID" = i."InvoiceID"  )  
	INNER JOIN "public"."CustomerPrice" c2 ON ( c2."Id" = c1."CustomerPriceID"  )  
	INNER JOIN "public"."Products" p ON ( p."Id" = c2."ProductID"  )  
WHERE p."ProductType" = 'Letter'   AND   to_date(cast(i."InvoiceDate" as TEXT),'YYYY-MM-DD')  >= '{{moment(DatePicker1.selectedDate).format("YYYY-MM-DD")}}' AND to_date(cast(i."InvoiceDate" as TEXT),'YYYY-MM-DD')  <=  '{{moment(DatePicker1.selectedDate).format("YYYY-MM-DD")}}'
